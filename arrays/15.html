                    int isPallindrome(int num){
                        int rev = 0, n = num;
                        while (num != 0)
                        {
                            rev = (rev*10) + (num%10);
                            num /= 10;
                        }
                        if (n == rev)
                        {
                            return 1;
                        }
                        return 0;
                    }
