            #define MAX_SIZE 100
            
            int main() {
                int arr[MAX_SIZE];
                int frequency[MAX_SIZE];
                int i, j, size;
            
                printf("Enter the elements of the array (maximum 100 elements, enter -1 to stop):\n");
                size = 0;
                int input;
                do {
                    scanf("%d", &input);
                    if (input != -1) {
                        arr[size] = input;
                        size++;
                    }
                } while (input != -1 && size < MAX_SIZE);
                for (i = 0; i < size; i++) {
                    frequency[i] = 1;
                }
                for (i = 0; i < size; i++) {
                    for (j = i + 1; j < size; j++) {
                        if (arr[i] == arr[j]) {
                            frequency[j] = 0;
                        }
                    }
                }
                printf("\nArray with duplicates removed and replaced by zero:\n");
                for (i = 0; i < size; i++) {
                    if (frequency[i] != 0) {
                        printf("%d ", arr[i]);
                    } else {
                        printf("0 ");
                    }
                }
            
                return 0;
            }
