#include <stdio.h>
    

                    // Function to calculate the Fibonacci sequence using recursion
                    unsigned long long fibonacci(int n) {
                        if (n <= 1) {
                            return n;
                        } else {
                            return fibonacci(n - 1) + fibonacci(n - 2);
                        }
                    }
                    
                    int main() {
                        int number;
                        unsigned long long result;
                    
                        printf("Enter a number to find its Fibonacci sequence: ");
                        scanf("%d", &number);
                    
                        if (number < 0) {
                            printf("Fibonacci sequence for a negative number is not defined.\n");
                        } else {
                            result = fibonacci(number);
                            printf("Fibonacci sequence at position %d = %llu\n", number, result);
                        }
                    
                        return 0;
                    }

output
