#include <stdio.h>
             

                    int sumOfDigits(int number) {
                        if (number == 0) {
                            return 0;
                        } else {
                            return (number % 10) + sumOfDigits(number / 10);
                        }
                    }
                    
                    int main() {
                        int num;
                        
                        printf("Enter a number: ");
                        scanf("%d", &num);
                        
                        int sum = sumOfDigits(num);
                        printf("The sum of the digits of %d is: %d\n", num, sum);
                        
                        return 0;
                    }

output

Enter a number: 2
                The sum of the digits of 2 is: 2
