#include <stdio.h>
                

                    // Function to convert a number to its binary equivalent using recursion
                    void decimalToBinary(int n) {
                        if (n > 1) {
                            decimalToBinary(n / 2);
                        }
                        printf("%d", n % 2);
                    }
                    
                    int main() {
                        int number;
                    
                        printf("Enter a number to convert to binary: ");
                        scanf("%d", &number);
                    
                        if (number < 0) {
                            printf("Please enter a non-negative number.\n");
                        } else {
                            printf("Binary equivalent: ");
                            if (number == 0) {
                                printf("0");
                            } else {
                                decimalToBinary(number);
                            }
                            printf("\n");
                        }
                    
                        return 0;
                    }

                Binary equivalent: 11
