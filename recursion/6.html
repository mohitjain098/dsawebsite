#include <stdio.h>
                

                    int findMin(int arr[], int n) {
                        
                        if (n == 1) {
                            return arr[0];
                        }
                        
                        
                        int minOfRest = findMin(arr, n - 1);
                    
                        
                        return (minOfRest < arr[n - 1]) ? minOfRest : arr[n - 1];
                    }
                    
                    int main() {
                        int array[] = {10, 25, 8, 30, 17};
                        int size = sizeof(array) / sizeof(array[0]);
                    
                        int min = findMin(array, size);
                        printf("The smallest element in the array is: %d\n", min);
                    
                        return 0;
                    }

output

The smallest element in the array is: 8
