                    void display(int mat[4][2]){
                        int i, j;
                        for (i = 0; i < 4; i++)

                            for (j = 0; j < 2; j++)
                            {
                                printf("%d\t", mat[i][j]);
                            }
                            printf("\n");
                        }
                    }
