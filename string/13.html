                    char* strrevcpy(char dest[], char src[]){
                        int i = 0, j = 0;
                        while(src[j] != '\0')
                            j++;
                        
                        while(src[i] != '\0'){
                            dest[i++] = src[--j];
                        } 
                        dest[i] = src[i]; 
                        return dest;
                    }
