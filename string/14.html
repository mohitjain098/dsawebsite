#include <stdio.h>
                

                    void printAsciiDifference(const char *str) {
                        int i = 0;
                        while (str[i] != '\0' && str[i + 1] != '\0') {
                            int diff = str[i + 1] - str[i];
                            printf("Difference between %c and %c: %d\n", str[i + 1], str[i], diff);
                            i++;
                        }
                    }
                    
                    int main() {
                        char inputString[100];
                    
                        printf("Enter a string: ");
                        fgets(inputString, sizeof(inputString), stdin);
                    
                        printf("Differences in ASCII values:\n");
                        printAsciiDifference(inputString);
                    
                        return 0;
                    }
