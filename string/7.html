                    int strCompare(char firststr[], char secondstr[]){
                        int i;
                        for (i = 0; firststr[i] != '\0' && firststr[i] != '\0'; i++){
                            if(firststr[i] != secondstr[i])
                            {
                                return 0;
                            }
                        }
                        if (firststr[i] != '\0' || secondstr[i] != '\0'){
                            return 0;
                        }
                        return 1;
                    }
