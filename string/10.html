#include <stdio.h>
               

                    int main() {
                        char inputString[100];
                        int index;
                    
                        printf("Enter a string: ");
                        fgets(inputString, sizeof(inputString), stdin);
                    
                        printf("Enter the index number: ");
                        scanf("%d", &index);
                    
                        // Check if the index is valid
                        if (index >= 0 && index < strlen(inputString)) {
                            printf("Character at index %d: %c\n", index, inputString[index]);
                        } else {
                            printf("Invalid index or out of range.\n");
                        }
                    
                        return 0;
                    }
